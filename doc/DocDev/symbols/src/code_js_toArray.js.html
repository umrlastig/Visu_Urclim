1 var Promise = require('any-promise')
  2

  9     done = stream

 11   }
 12

 18

 20

 22     stream.on('end', onEnd)
 23     stream.on('error', onEnd)
 24     stream.on('close', onClose)

 26     function onData(doc) {
 27       arr.push(doc)
 28     }
 29 
 30     function onEnd(err) {
 31       if (err) reject(err)
 32       else resolve(arr)
 33       cleanup()
 34     }
 35 
 36     function onClose() {
 37       resolve(arr)
 38       cleanup()
 39     }
 40 
 41     function cleanup() {
 42       arr = null
 43       stream.removeListener('data', onData)
 44       stream.removeListener('end', onEnd)
 45       stream.removeListener('error', onEnd)
 46       stream.removeListener('close', onClose)
 47     }
 48   })
 49 
 50   if (typeof done === 'function') {
 51     deferred.then(function (arr) {
 52       process.nextTick(function() {
 53         done(null, arr)
 54       })
 55     }, done)
 56   }
 57 
 58   return deferred
 59 }
