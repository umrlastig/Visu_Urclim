  5 /**
  6  * Cette fonction permet de charger le bati et la route pour un jeu de donnée et une emprise précise
  7  */
  8 export function chargeBatiEtRoute() {
  9   
 10 	//Si aucun jeu de donnée n'est selectionné, on prend celle de paris_centre
 11 	var dataset = "paris_centre";
 12 	var road_path = "";
 13 	var build_path = "";
 14 	
 15 	if ($("#data_road").val() != "" ){
 16 		road_path = "geojson/" + $("#data_road").val().split("fakepath\\")[1];
 17 	}
 18 	else{
 19 		road_path = "geojson/roads_" + dataset + ".geojson";
 20 	}
 21 
 22 	if ($("#data_build").val() != ""){
 23 		build_path = "geojson/" + $("#data_build").val().split("fakepath\\")[1];
 24 	}
 25 	else{
 26 		build_path = "geojson/buildings_" + dataset + ".geojson";
 27 	}
 28 
 29 	fetch(road_path).then(r => r.json()).then(function (data) {
 30 		general_config.data_road = data;
 31 		filtreRoute();
 32 	});
 33 
 34 	fetch(build_path).then(r => r.json()).then(function (data) {
 35 		general_config.data_build = data;
 36 		filtreBati();
 37 		setTimeout(function (d) {
 38 			create_buildings(general_config.data_build, scene, $("#type_bati").val());
 39 		}, 1000);
 40 	});
 41 
 42 	$("#type_bati").on("change", function () {
 43 		create_buildings(general_config.data_build, scene, $("#type_bati").val());
 44 	});
 45 }
 46 /**
 47  * Cette fonction filtre le bati pour ne garder que la zone voulue
 48  */
 49 export function filtreBati() {
 50 
 51 	var features = general_config.data_build.features;
 52 	var res = [];
 53 	features.forEach(bati => {
 54 		var sommeX = 0;
 55 		var sommeY = 0;
 56 		var n = 0;
 57 		bati.geometry.coordinates.forEach(tab => {
 58 			tab.forEach(points => {
 59 				points.forEach(point => {
 60 					sommeX += point[0];
 61 					sommeY += point[1];
 62 					n += 1;
 63 				});
 64 			});
 65 		});
 66 		var centreX = sommeX / n;
 67 		var centreY = sommeY / n;
 68 		if (centreX > general_config.x_min && centreX < general_config.x_max && centreY > general_config.y_min && centreY < general_config.y_max) {
 69 			res.push(bati);
 70 		}
 71 	});
 72 	general_config.data_build.features = res;
 73 }

102 /**
103  * Cette fonction filtre les points O et u,v pour ne garder que la zone voulue
104 	Il ne garde que les points dans les colonnes et les lignes du quadrillage
105 	Pour prévenir le probleme : un point d'une ligne ne fait pas partis de l'emprise mais toute les autres en fond partis
106  * @param {*} data_list 
107  */
108 export function filtreDonneeOUV(data_list){
109 
110     var featuresO = data_list.listePoints;
111     var featuresU = data_list.listePointsU;
112     var featuresV = data_list.listePointsV;
113     var resO = [];
114     var resU = [];
115     var resV = [];
116     for (let i = 0; i < featuresO.length; i++) {
117         const point = featuresO[i];
118         if (point.i_point >= data_list.globalData.ni_min && point.i_point <= data_list.globalData.ni_max && point.j_point >= data_list.globalData.nj_min && point.j_point <= data_list.globalData.nj_max){
119             resO.push(point);
120             resU.push(featuresU[i]);
121             resV.push(featuresV[i]);
122         }
123     }
124     data_list.listePoints = resO;
125     data_list.listePointsU = resU;
126     data_list.listePointsV = resV;
127 }

129 /**
130  * Cette fonction recalcule les températures minimales et maximales
131  * @param {*} data_list 
132  */
133 export function calcTempMinMax(data_list){
134 	let temp_max = 0;
135 	let temp_min = 999;
136 
137 	//On cherche les temp_min et max du fichier
138 	data_list.listePoints.forEach((point) => {
139 		for (let key in point) {
140 			if (
141 				(key.startsWith("tht") ||
142 					(key.startsWith("teb") && !key.startsWith("tebz"))) &&
143 				point[key] != 999
144 			) {
145 				if (point[key] > temp_max) {
146 					temp_max = point[key];
147 				}
148 				if (point[key] < temp_min) {
149 					temp_min = point[key];
150 				}
151 			}
152 		}
153 	});
154 
155 	//On recalcul general_config.temp_min et general_config.temp_max
156 	if (temp_min < general_config.temp_min){
157 		general_config.temp_min = temp_min;
158 	}
159 	if (temp_max > general_config.temp_max){
160 		general_config.temp_max = temp_max;
161 	}
162 
163 }

165 /**
166  * Recalcule nombre ligne et colonne
167  * @param {*} data_list 
168  */
169 export function findNiNj(data_list){
170 
171 	let ni_min;
172 	let nj_min;
173 	if(data_list.globalData.ni_max){
174 		ni_min = data_list.globalData.ni_max;
175 		nj_min = data_list.globalData.nj_max;
176 	}else{
177 		ni_min = data_list.globalData.ni;
178 		nj_min = data_list.globalData.nj;
179 	}
180 	
181 	var ni_max = 0;
182 	
183 	var nj_max = 0;
184 	for (let index = 0; index < data_list.listePoints.length; index++) {
185 		const point_actuel = data_list.listePoints[index];
186 		if (point_actuel.latitude >= general_config.lat_min && point_actuel.latitude <= general_config.lat_max && point_actuel.longitude >= general_config.lng_min && point_actuel.longitude <= general_config.lng_max){
187 			if (point_actuel.i_point < ni_min){
188 				ni_min=point_actuel.i_point;
189 			}
190 			if (point_actuel.i_point > ni_max){
191 				ni_max=point_actuel.i_point;
192 			}
193 			
194 			if (point_actuel.j_point < nj_min){
195 				nj_min=point_actuel.j_point;
196 			}
197 			if (point_actuel.j_point > nj_max){
198 				nj_max=point_actuel.j_point;
199 			}
200 		}
201 	}
202 	data_list.globalData.ni = ni_max - ni_min + 1;
203 	data_list.globalData.nj = nj_max - nj_min + 1;
204 	data_list.globalData.ni_min = ni_min;
205 	data_list.globalData.ni_max = ni_max;
206 	data_list.globalData.nj_min = nj_min;
207 	data_list.globalData.nj_max = nj_max;
208 }

210 /**
211  * Cette fonction permet de recalculer l'emprise des bati pour qu'elle s'adapte aux plaques
212  * @param {*} data_list 
213  */
214 export function newFootprint(data_list){
215 
216 	let ni = data_list.globalData.ni;
217 	let nj = data_list.globalData.nj;
218 	let pointS0 = data_list.listePoints[0];
219 	let pointNE = data_list.listePoints[ni*nj-1];
220 	if (ni == 1 || nj ==1){
221 		var deltaLong = 0.004266142845153809;
222 		var deltaLat = 0.0028076171875;
223 	}else{
224 		var deltaLong = Math.abs(data_list.listePoints[0].longitude - data_list.listePoints[1].longitude)/2;
225 		var deltaLat = Math.abs(data_list.listePoints[0].latitude - data_list.listePoints[ni].latitude)/2;
226 
227 	}
228 	
229 	
230 	general_config.lat_min = pointS0.latitude-deltaLat;
231 	general_config.lat_max = pointNE.latitude+deltaLat;
232 	general_config.lng_min = pointS0.longitude-deltaLong;
233 	general_config.lng_max = pointNE.longitude+deltaLong;
234 	var EPSG2154 =
235 		"+proj=lcc +lat_1=49 +lat_2=44 +lat_0=46.5 +lon_0=3 +x_0=700000 +y_0=6600000 +ellps=GRS80 +towgs84=0,0,0,0,0,0,0 +units=m +no_defs";
236 	reprojetteCoordMinMax(EPSG2154);
237 
238 	general_config.Coord_X_paris = (general_config.x_min+general_config.x_max)/2;
239 	general_config.Coord_Y_paris = (general_config.y_min+general_config.y_max)/2;
240 };
