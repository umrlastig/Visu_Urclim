1 // Création de la carte leaflet pour le choix de l'emprise
  2 export var carte = L.map('footprint_map').setView([48.8534, 2.3488], 12);
  3 L.tileLayer('http://{s}.tile.osm.org/{z}/{x}/{y}.png', {
  4     attribution: '© <a href="http://osm.org/copyright">OpenStreetMap</a> contributors'
  5     }).addTo(carte);
  6 
  7 //style marqueur
  8 var icon_marqueur = L.icon({
  9     iconUrl: './images/emprise.jpg',
 10     iconSize: [10, 10],
 11 
 12 })
 13 // création des layers hors de la fonction pour pouvoir les manipuler plus facilement 
 14 export var marker_front1 = L.marker([0,0], {draggable:'true', 'opacity': 0, icon: icon_marqueur}).addTo(carte);
 15 export var marker_front2 = L.marker([0,0], {draggable:'true' , 'opacity': 0, icon: icon_marqueur}).addTo(carte);
 16 export var polygone = L.rectangle([[0,0],[0,0],[0,0],[0,0]], {'opacity' :0}).addTo(carte);
 17 export var groupLayer = L.layerGroup();
 18 
 19 /**
 20  * Cette fonction permet de créer un polygone dynamique ainsi que de d'afficher les points des fichiers
 21  * @param {*} carte 
 22  * @param {*} lat_min 
 23  * @param {*} lat_max 
 24  * @param {*} lng_min 
 25  * @param {*} lng_max 
 26  * @param {*} listepoint 
 27  */
 28 export function display_footprint(carte,lat_min,lat_max,lng_min,lng_max,listepoint){
 29 
 30     let lat_moy = (lat_min+lat_max)/2;
 31     let lng_moy = (lng_min+lng_max)/2;
 32     carte.setView(new L.LatLng(lat_moy, lng_moy));
 33     var bounds = [[lat_min, lng_min], [lat_max, lng_max]];
 34     marker_front1.setLatLng([lat_min, lng_min]);
 35     marker_front1.setOpacity(1);
 36     marker_front2.setLatLng([lat_max, lng_max]);
 37     marker_front2.setOpacity(1);
 38 
 39 
 40     polygone.setBounds(bounds);
 41     polygone.setStyle({
 42     'color': '#FF7F00',
 43     'fill': true,
 44     'fillColor': '#FFFF00',
 45     'fillOpacity': 0.2,
 46     'opacity' : 1
 47    });
 48     marker_front1.on('drag',() => deplacementCurseur(lat_min,lat_max,lng_min,lng_max));
 49     marker_front2.on('drag',() => deplacementCurseur(lat_min,lat_max,lng_min,lng_max));    
 50 
 51     
 52 
 53     //netoyage des point précédents
 54     groupLayer.eachLayer(function (layer) {
 55         groupLayer.removeLayer(layer);
 56     });
 57     groupLayer.addTo(carte);
 58     //apparition des points
 59     listepoint.forEach(point => {
 60         groupLayer.addLayer(L.circleMarker([point.latitude, point.longitude], {color: 'blue', radius: 5}));
 61         groupLayer.addTo(carte);
 62     });
 63 
 64     /**
 65      * Cette méthode permet de gérer le deplacement d'un des curseurs, en modifiant les lat et long min et max et en changeant la couleur des points
 66      * @param {*} lat_min 
 67      * @param {*} lat_max 
 68      * @param {*} lng_min 
 69      * @param {*} lng_max 
 70      */
 71     function deplacementCurseur(lat_min,lat_max,lng_min,lng_max) {
 72         let coord1 = marker_front1.getLatLng();
 73         let coord2 = marker_front2.getLatLng();
 74         // vérification que le curseur 1 ne sorte pas de l'emprise max
 75         if (coord1['lat']<lat_min){
 76             coord1['lat']=lat_min;
 77         }else if(coord1['lat']>lat_max){
 78             coord1['lat']=lat_max;
 79         }
 80         if (coord1['lng']<lng_min){
 81             coord1['lng']=lng_min;
 82         }else if(coord1['lng']>lng_max){
 83             coord1['lng']=lng_max;
 84         }
 85         // vérification que le curseur 2 ne sorte pas de l'emprise max
 86         if (coord2['lat']<lat_min){
 87             coord2['lat']=lat_min;
 88         }else if(coord2['lat']>lat_max){
 89             coord2['lat']=lat_max;
 90         }
 91         if (coord2['lng']<lng_min){
 92             coord2['lng']=lng_min;
 93         }else if(coord2['lng']>lng_max){
 94             coord2['lng']=lng_max;
 95         }
 96 
 97         let new_bounds = L.latLngBounds(coord1, coord2);
 98         polygone.setBounds(new_bounds);
 99 
100         // changement de couleur 
101         groupLayer.eachLayer(function (layer) {
102             let coordPoint = layer.getLatLng();
103             if(new_bounds.contains(coordPoint)){
104                 layer.setStyle({color: 'blue', radius: 5});
105             }else{
106                 layer.setStyle({color: 'red', radius: 5});
107             }
108         });        
109     };      
110     
111 };
