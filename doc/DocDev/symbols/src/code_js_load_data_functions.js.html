353 /**
354  * permet de recuperer les coordonnees de l'emprise selectionnée par l'utilisateur.
355  * @param {*} marker_front1 
356  * @param {*} marker_front2 
357  * @param {*} data_to_load_list 
358  */
359 export function selectfootprint(marker_front1, marker_front2,data_to_load_list) {
360 
361 	let coord1 = marker_front1.getLatLng();
362 	let coord2 = marker_front2.getLatLng();
363 
364 	// vérification que le curseur 1 ne sorte pas de l'emprise max
365 	if (coord1["lat"] < data_to_load_list.globalData.latitude_min) {
366 		coord1["lat"] = data_to_load_list.globalData.latitude_min;
367 	} else if (coord1["lat"] > data_to_load_list.globalData.latitude_max) {
368 		coord1["lat"] = data_to_load_list.globalData.latitude_max;
369 	}
370 	if (coord1["lng"] < data_to_load_list.globalData.longitude_min) {
371 		coord1["lng"] = data_to_load_list.globalData.longitude_min;
372 	} else if (coord1["lng"] > data_to_load_list.globalData.longitude_max) {
373 		coord1["lng"] = data_to_load_list.globalData.longitude_max;
374 	}
375 	// vérification que le curseur 2 ne sorte pas de l'emprise max
376 	if (coord2["lat"] < data_to_load_list.globalData.latitude_min) {
377 		coord2["lat"] = data_to_load_list.globalData.latitude_min;
378 	} else if (coord1["lat"] > data_to_load_list.globalData.latitude_max) {
379 		coord2["lat"] = data_to_load_list.globalData.latitude_max;
380 	}
381 	if (coord2["lng"] < data_to_load_list.globalData.longitude_min) {
382 		coord2["lng"] = data_to_load_list.globalData.longitude_min;
383 	} else if (coord2["lng"] > data_to_load_list.globalData.longitude_max) {
384 		coord2["lng"] = data_to_load_list.globalData.longitude_max;
385 	}
386 
387 	if (coord1["lat"] <= coord2["lat"]) {
388 		general_config.lat_min = coord1["lat"];
389 		general_config.lat_max = coord2["lat"];
390 	} else {
391 		general_config.lat_min = coord2["lat"];
392 		general_config.lat_max = coord1["lat"];
393 	}
394 	if (coord1["lng"] <= coord2["lng"]) {
395 		general_config.lng_min = coord1["lng"];
396 		general_config.lng_max = coord2["lng"];
397 	} else {
398 		general_config.lng_min = coord2["lng"];
399 		general_config.lng_max = coord1["lng"];
400 	}
401 
402 	showFootprint();
403 	showData();
404 
405 	//Cette fonction doit etre lancer lorsque tout les fichiers sont charger et que l'on a cliqué sur select footprint
406 	// Il faut donc changer le temps, ou trouver une meilleure méthode plus performante
407 	loadAll();
408 	
409 	//on redessine l'emprise pour enlever les points hors de l'emprise selectionné
410 	display_footprint(carte, 
411         general_config.lat_min, 
412         general_config.lat_max, 
413         general_config.lng_min, 
414         general_config.lng_max, 
415         general_config.netcdf_list[0].listePoints);
416 
417 }
418 /**
419  * Après selection de l'emprise, cette méthode charge tout : bati, OUV, et reprojette
420  */
421 function loadAll() {
422 
423 
424 	var EPSG2154 = "+proj=lcc +lat_1=49 +lat_2=44 +lat_0=46.5 +lon_0=3 +x_0=700000 +y_0=6600000 +ellps=GRS80 +towgs84=0,0,0,0,0,0,0 +units=m +no_defs";
425 	reprojetteCoordMinMax(EPSG2154);
426 
427 	general_config.temp_min = 999;
428 	general_config.temp_max = 0;
429 
430 	general_config.netcdf_list.forEach(data_list => {
431 		findNiNj(data_list);
432 		filtreDonneeOUV(data_list);
433 		reprojette(EPSG2154);
434 		calcTempMinMax(data_list);
435 	});
436 	
437 	general_config.temp_array = [general_config.temp_min,general_config.temp_max];
438 	general_config.domain_max = Math.floor(general_config.temp_max + 1 - 273.15);
439 	general_config.domain_min = Math.floor(general_config.temp_min - 273.15);
440 
441 	newFootprint(general_config.netcdf_list[0]);
442 	chargeBatiEtRoute();
443 	setCurrent(0);
444 	draw_graph2D();
445     showTimeVis(general_config.netcdf_list);
446 }
447 /**
448  * Cette fonction permet de calculer les valeurs X/Y min/max après calcul d'emprise
449  * @param {*} projection 
450  */
451 export function reprojetteCoordMinMax(projection) {
452 
453 	var res1 = proj4(projection, [
454 		general_config.lng_min,
455 		general_config.lat_min,
456 	]);
457 	var res2 = proj4(projection, [
458 		general_config.lng_max,
459 		general_config.lat_max,
460 	]);
461 	general_config.x_min = res1[0];
462 	general_config.x_max = res2[0];
463 	general_config.y_min = res1[1];
464 	general_config.y_max = res2[1];
465 }

467 /**
468  * Cette fonction permet de calculer les valeurs X/Y après calcul d'emprise
469  * @param {*} data 
470  * @param {*} projection 
471  */
472 function reprojetteCoordOUV(data, projection) {
473 
474 	data.forEach((point) => {
475 		try {
476 			var res = proj4(projection, [point.longitude, point.latitude]);
477 			point.x = res[0];
478 			point.y = res[1];
479 		} catch (error) {
480 			console.log(error);
481 		}
482 	});
483 }
484 /**
485  * lance reprojetteCoordMinMax et reprojetteCoordOUV pour tout les fichiers
486  * @param {*} projection 
487  */
488 function reprojette(projection) {
489 
490 	general_config.netcdf_list.forEach(data_list => {
491 		reprojetteCoordOUV(data_list.listePoints, projection);
492 		reprojetteCoordOUV(data_list.listePointsU, projection);
493 		reprojetteCoordOUV(data_list.listePointsV, projection);
494 	});
495 }
