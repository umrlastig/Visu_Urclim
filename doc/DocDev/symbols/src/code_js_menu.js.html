255 /**
256  * Cette fonction permet de remplir et trier la liste des dates affichée dans le panneau Data du menu.
257  * @param {*} netcdfList 
258  * @param {*} dateSelect 
259  */
260 function sortDate(netcdfList,dateSelect){
261   var dateArray = new Array();
262     for (let i=0;i<netcdfList.length;i++) {
263         dateArray[i] = new Date(netcdfList[i].globalData.date).getTime();
264     }
265     dateArray.sort();
266     dateSelect.options.length = 0;
267 
268     for (let i=0;i<dateArray.length;i++) {
269       dateSelect.options[i] = new Option(new Date(dateArray[i]));
270     }
271 }

274 /**
275  * This function update the data displayed on the main pannel of the app when the user change the date in the data pannel.
276  * @param {*} date 
277  */
278 export function changeNetcdf(date) {
279   for (let i = 0; i < general_config.netcdf_list.length; i++) {
280     if (general_config.netcdf_list[i].globalData.date == date) {
281       setCurrent(i);
282       break;
283     }
284   }
285 
286   //horizontal plane checkboxes
287   let checks = document.querySelectorAll("div.input_data_container_1 > input");
288   for (let i = 0; i < checks.length; i++) {
289     let check = checks[i];
290     if (check.checked) {
291       scene.remove(general_config.grid_plane);
292       general_config.grid_plane = new THREE.Object3D();
293       create_2D_plane_series(
294         general_config.data_points_O_2,
295         general_config.data_points_U_2,
296         general_config.data_points_V_2,
297         general_config.grid_plane,
298         general_config.id_sbl_array_real_plane,
299         general_config.id_meso_array_real_plane,
300         general_config.temp_array,
301         general_config.THAT,
302         general_config.THAT_W,
303         general_config.HCanopy,
304         general_config.HCanopy_w
305       );
306       create_temp_histogram();
307       break;
308     }
309   }
310 
311   //cloud point checkboxes
312   let checks2 = document.querySelectorAll("div.input_data_container_2 > input");
313   for (var i = 0; i < checks2.length; i++) {
314     var check = checks2[i];
315     if (!check.disabled) {
316       if (check.checked) {
317 
318         scene.remove(general_config.grid);
319         general_config.grid = new THREE.Object3D();
320         chooseCreationPoints();
321         break;
322       }
323       check.checked = false;
324     }
325   }
326 
327   //vertical checkboxes
328   let checks3 = document.querySelectorAll("div.input_data_container_3 > input");
329   for (let i = 0; i < checks3.length; i++) {
330     let check = checks3[i];
331     if (check.checked) {
332       scene.remove(general_config.grid);
333       general_config.grid_plane = new THREE.Object3D();
334       create_2D_vertical_plane_series(
335         general_config.data_road,
336         general_config.grid_vertical2D,
337         general_config.id_sbl_array_vertical_plane,
338         general_config.id_meso_array_vertical_plane,
339         general_config.temp_array,
340         general_config.THAT,
341         general_config.THAT_W,
342         general_config.HCanopy,
343         general_config.HCanopy_w
344       );
345       create_temp_histogram();
346       break;
347     }
348   }
349 }

351 /**
352  * Cette fonction permet de switcher automatiquement de fichier en
353  * prenant celui à la date suivante toutes les 3 secondes
354  */
355 function activateTemporalAnimation(){
356   let temporal_animation = document.getElementById('temporal_animation');
357   let dateControl = document.getElementById('date_control_input');
358   let timeStep = document.getElementById('temporal_animation_time_step').value * 1000;
359   if(timeStep < 1000){
360     timeStep = 1000;
361   }
362     setTimeout(()=>{
363       if(temporal_animation.checked){
364         let index = dateControl.selectedIndex + 1;
365         if(index >= dateControl.length){
366           index = 0;
367         }
368         dateControl.options[index].selected = true;
369         changeNetcdf(dateControl.value);
370         updateCurrentSelected3D()
371         activateTemporalAnimation();
372       }      
373     },timeStep);
374 }

376 /**
377  * prevent radio button swap when using keyboard
378  */
379 export function preventRadioSwap(){
380   $('input[type="radio"]').keydown(function(e)
381   {
382       var arrowKeys = [37, 38, 39, 40];
383       if (arrowKeys.indexOf(e.which) !== -1)
384       {
385           $(this).blur();
386           return false;
387       }
388   });
389 }
