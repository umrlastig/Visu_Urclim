 19 // This set of controls performs orbiting, dollying (zooming), and panning.
 20 // Unlike TrackballControls, it maintains the "up" direction object.up (+Y by default).
 21 //
 22 //    Orbit - left mouse / touch: one-finger move
 23 //    Zoom - middle mouse, or mousewheel / touch: two-finger spread or squish
 24 //    Pan - right mouse, or left mouse + ctrl/metaKey, or arrow keys / touch: two-finger move
 25 
 26 var OrbitControls = function ( object, domElement ) {
 27 
 28 	this.object = object;
 29 
 30 	this.domElement = ( domElement !== undefined ) ? domElement : document;
 31 
 32 	// Set to false to disable this control
 33 	this.enabled = true;
 34 
 35 	// "target" sets the location of focus, where the object orbits around
 36 	this.target = new Vector3();
 37 
 38 	// How far you can dolly in and out ( PerspectiveCamera only )
 39 	this.minDistance = 0;
 40 	this.maxDistance = Infinity;
 41 
 42 	// How far you can zoom in and out ( OrthographicCamera only )
 43 	this.minZoom = 0;
 44 	this.maxZoom = Infinity;
 45 
 46 	// How far you can orbit vertically, upper and lower limits.
 47 	// Range is 0 to Math.PI radians.
 48 	this.minPolarAngle = 0; // radians
 49 	this.maxPolarAngle = Math.PI; // radians
 50 
 51 	// How far you can orbit horizontally, upper and lower limits.
 52 	// If set, must be a sub-interval of the interval [ - Math.PI, Math.PI ].
 53 	this.minAzimuthAngle = - Infinity; // radians
 54 	this.maxAzimuthAngle = Infinity; // radians
 55 
 56 	// Set to true to enable damping (inertia)
 57 	// If damping is enabled, you must call controls.update() in your animation loop
 58 	this.enableDamping = false;
 59 	this.dampingFactor = 0.25;
 60 
 61 	// This option actually enables dollying in and out; left as "zoom" for backwards compatibility.
 62 	// Set to false to disable zooming
 63 	this.enableZoom = true;
 64 	this.zoomSpeed = 1.0;
 65 
 66 	// Set to false to disable rotating
 67 	this.enableRotate = true;
 68 	this.rotateSpeed = 1.0;
 69 
 70 	// Set to false to disable panning
 71 	this.enablePan = true;
 72 	this.panSpeed = 1.0;
 73 	this.screenSpacePanning = false; // if true, pan in screen-space
 74 	this.keyPanSpeed = 7.0;	// pixels moved per arrow key push
 75 
 76 	// Set to true to automatically rotate around the target
 77 	// If auto-rotate is enabled, you must call controls.update() in your animation loop
 78 	this.autoRotate = false;
 79 	this.autoRotateSpeed = 2.0; // 30 seconds per round when fps is 60
 80 
 81 	// Set to false to disable use of the keys
 82 	this.enableKeys = true;
 83 
 84 	// The four arrow keys
 85 	this.keys = { LEFT: 37, UP: 38, RIGHT: 39, BOTTOM: 40 };
 86 
 87 	// Mouse buttons
 88 	this.mouseButtons = { LEFT: MOUSE.LEFT, MIDDLE: MOUSE.MIDDLE, RIGHT: MOUSE.RIGHT };
 89 
 90 	// for reset
 91 	this.target0 = this.target.clone();
 92 	this.position0 = this.object.position.clone();
 93 	this.zoom0 = this.object.zoom;

134 	// this method is exposed, but perhaps it would be better if we can make it private...
135 	this.update = function () {
136 
137 		var offset = new Vector3();
138 
139 		// so camera.up is the orbit axis
140 		var quat = new Quaternion().setFromUnitVectors( object.up, new Vector3( 0, 1, 0 ) );
141 		var quatInverse = quat.clone().inverse();
142 
143 		var lastPosition = new Vector3();
144 		var lastQuaternion = new Quaternion();
145 
146 		return function update() {
147 
148 			var position = scope.object.position;
149 
150 			offset.copy( position ).sub( scope.target );
151 
152 			// rotate offset to "y-axis-is-up" space
153 			offset.applyQuaternion( quat );
154 
155 			// angle from z-axis around y-axis
156 			spherical.setFromVector3( offset );
157 
158 			if ( scope.autoRotate && state === STATE.NONE ) {
159 
160 				rotateLeft( getAutoRotationAngle() );
161 
162 			}
163 
164 			spherical.theta += sphericalDelta.theta;
165 			spherical.phi += sphericalDelta.phi;
166 
167 			// restrict theta to be between desired limits
168 			spherical.theta = Math.max( scope.minAzimuthAngle, Math.min( scope.maxAzimuthAngle, spherical.theta ) );
169 
170 			// restrict phi to be between desired limits
171 			spherical.phi = Math.max( scope.minPolarAngle, Math.min( scope.maxPolarAngle, spherical.phi ) );
172 
173 			spherical.makeSafe();
174 
175 
176 			spherical.radius *= scale;
177 
178 			// restrict radius to be between desired limits
179 			spherical.radius = Math.max( scope.minDistance, Math.min( scope.maxDistance, spherical.radius ) );
180 
181 			// move target to panned location
182 			scope.target.add( panOffset );
183 
184 			offset.setFromSpherical( spherical );
185 
186 			// rotate offset back to "camera-up-vector-is-up" space
187 			offset.applyQuaternion( quatInverse );
188 
189 			position.copy( scope.target ).add( offset );
190 
191 			scope.object.lookAt( scope.target );
192 
193 			if ( scope.enableDamping === true ) {
194 
195 				sphericalDelta.theta *= ( 1 - scope.dampingFactor );
196 				sphericalDelta.phi *= ( 1 - scope.dampingFactor );
197 
198 				panOffset.multiplyScalar( 1 - scope.dampingFactor );
199 
200 			} else {
201 
202 				sphericalDelta.set( 0, 0, 0 );
203 
204 				panOffset.set( 0, 0, 0 );
205 
206 			}
207 
208 			scale = 1;
209 
210 			// update condition is:
211 			// min(camera displacement, camera rotation in radians)^2 > EPS
212 			// using small-angle approximation cos(x/2) = 1 - x^2 / 8
213 
214 			if ( zoomChanged ||
215 				lastPosition.distanceToSquared( scope.object.position ) > EPS ||
216 				8 * ( 1 - lastQuaternion.dot( scope.object.quaternion ) ) > EPS ) {
217 
218 				scope.dispatchEvent( changeEvent );
219 
220 				lastPosition.copy( scope.object.position );
221 				lastQuaternion.copy( scope.object.quaternion );
222 				zoomChanged = false;
223 
224 				return true;
225 
226 			}
227 
228 			return false;
229 
230 		};
231 
232 	}();

313 	var panLeft = function () {
314 
315 		var v = new Vector3();
316 
317 		return function panLeft( distance, objectMatrix ) {
318 
319 			v.setFromMatrixColumn( objectMatrix, 0 ); // get X column of objectMatrix
320 			v.multiplyScalar( - distance );
321 
322 			panOffset.add( v );
323 
324 		};

328 	var panUp = function () {
329 
330 		var v = new Vector3();
331 
332 		return function panUp( distance, objectMatrix ) {
333 
334 			if ( scope.screenSpacePanning === true ) {
335 
336 				v.setFromMatrixColumn( objectMatrix, 1 );
337 
338 			} else {
339 
340 				v.setFromMatrixColumn( objectMatrix, 0 );
341 				v.crossVectors( scope.object.up, v );
342 
343 			}
344 
345 			v.multiplyScalar( distance );
346 
347 			panOffset.add( v );
348 
349 		};

353 	// deltaX and deltaY are in pixels; right and down are positive
354 	var pan = function () {
355 
356 		var offset = new Vector3();
357 
358 		return function pan( deltaX, deltaY ) {
359 
360 			var element = scope.domElement === document ? scope.domElement.body : scope.domElement;
361 
362 			if ( scope.object.isPerspectiveCamera ) {
363 
364 				// perspective
365 				var position = scope.object.position;
366 				offset.copy( position ).sub( scope.target );
367 				var targetDistance = offset.length();
368 
369 				// half of the fov is center to top of screen
370 				targetDistance *= Math.tan( ( scope.object.fov / 2 ) * Math.PI / 180.0 );
371 
372 				// we use only clientHeight here so aspect ratio does not distort speed
373 				panLeft( 2 * deltaX * targetDistance / element.clientHeight, scope.object.matrix );
374 				panUp( 2 * deltaY * targetDistance / element.clientHeight, scope.object.matrix );
375 
376 			} else if ( scope.object.isOrthographicCamera ) {
377 
378 				// orthographic
379 				panLeft( deltaX * ( scope.object.right - scope.object.left ) / scope.object.zoom / element.clientWidth, scope.object.matrix );
380 				panUp( deltaY * ( scope.object.top - scope.object.bottom ) / scope.object.zoom / element.clientHeight, scope.object.matrix );
381 
382 			} else {
383 
384 				// camera neither orthographic nor perspective
385 				console.warn( 'WARNING: OrbitControls.js encountered an unknown camera type - pan disabled.' );
386 				scope.enablePan = false;
387 
388 			}
389 
390 		};
391 
392 	}();

394 	function dollyIn( dollyScale ) {
395 
396 		if ( scope.object.isPerspectiveCamera ) {
397 
398 			scale /= dollyScale;
399 
400 		} else if ( scope.object.isOrthographicCamera ) {
401 
402 			scope.object.zoom = Math.max( scope.minZoom, Math.min( scope.maxZoom, scope.object.zoom * dollyScale ) );
403 			scope.object.updateProjectionMatrix();
404 			zoomChanged = true;
405 
406 		} else {
407 
408 			console.warn( 'WARNING: OrbitControls.js encountered an unknown camera type - dolly/zoom disabled.' );
409 			scope.enableZoom = false;
410 
411 		}
412 
413 	}
414 
415 	function dollyOut( dollyScale ) {
416 
417 		if ( scope.object.isPerspectiveCamera ) {
418 
419 			scale *= dollyScale;
420 
421 		} else if ( scope.object.isOrthographicCamera ) {
422 
423 			scope.object.zoom = Math.max( scope.minZoom, Math.min( scope.maxZoom, scope.object.zoom / dollyScale ) );
424 			scope.object.updateProjectionMatrix();
425 			zoomChanged = true;
426 
427 		} else {
428 
429 			console.warn( 'WARNING: OrbitControls.js encountered an unknown camera type - dolly/zoom disabled.' );
430 			scope.enableZoom = false;
431 
432 		}
433 
434 	}

694 	function onMouseDown( event ) {
695 
696 		if ( scope.enabled === false ) return;
697 
698 		event.preventDefault();
699 
700 		switch ( event.button ) {
701 
702 			case scope.mouseButtons.LEFT:
703 
704 				if ( event.ctrlKey || event.metaKey ) {
705 
706 					if ( scope.enablePan === false ) return;
707 
708 					handleMouseDownPan( event );
709 
710 					state = STATE.PAN;
711 
712 				} else {
713 
714 					if ( scope.enableRotate === false ) return;
715 
716 					handleMouseDownRotate( event );
717 
718 					state = STATE.ROTATE;
719 
720 				}
721 
722 				break;
723 
724 			case scope.mouseButtons.MIDDLE:
725 
726 				if ( scope.enableZoom === false ) return;
727 
728 				handleMouseDownDolly( event );
729 
730 				state = STATE.DOLLY;
731 
732 				break;
733 
734 			case scope.mouseButtons.RIGHT:
735 
736 				if ( scope.enablePan === false ) return;
737 
738 				handleMouseDownPan( event );
739 
740 				state = STATE.PAN;
741 
742 				break;
743 
744 		}
745 
746 		if ( state !== STATE.NONE ) {
747 
748 			document.addEventListener( 'mousemove', onMouseMove, false );
749 			document.addEventListener( 'mouseup', onMouseUp, false );
750 
751 			scope.dispatchEvent( startEvent );
752 
753 		}
754 
755 	}

758 
759 		if ( scope.enabled === false ) return;
760 
761 		event.preventDefault();
762 
763 		switch ( state ) {
764 
765 			case STATE.ROTATE:
766 
767 				if ( scope.enableRotate === false ) return;
768 
769 				handleMouseMoveRotate( event );
770 
771 				break;
772 
773 			case STATE.DOLLY:
774 
775 				if ( scope.enableZoom === false ) return;
776 
777 				handleMouseMoveDolly( event );
778 
779 				break;
780 
781 			case STATE.PAN:
782 
783 				if ( scope.enablePan === false ) return;
784 
785 				handleMouseMovePan( event );
786 
787 				break;
788 
789 		}

793 	function onMouseUp( event ) {
794 
795 		if ( scope.enabled === false ) return;
796 
797 		handleMouseUp( event );
798 
799 		document.removeEventListener( 'mousemove', onMouseMove, false );
800 		document.removeEventListener( 'mouseup', onMouseUp, false );
801 
802 		scope.dispatchEvent( endEvent );
803 
804 		state = STATE.NONE;
805 
806 	}

808 	function onMouseWheel( event ) {
809 
810 		if ( scope.enabled === false || scope.enableZoom === false || ( state !== STATE.NONE && state !== STATE.ROTATE ) ) return;
811 
812 		event.preventDefault();
813 		event.stopPropagation();
814 
815 		scope.dispatchEvent( startEvent );
816 
817 		handleMouseWheel( event );
818 
819 		scope.dispatchEvent( endEvent );
820 
821 	}

828 
829 	}

831 	function onTouchStart( event ) {
832 
833 		if ( scope.enabled === false ) return;
834 
835 		// event.preventDefault();  // comment out @minorua
836 
837 		switch ( event.touches.length ) {
838 
839 			case 1:	// one-fingered touch: rotate
840 
841 				if ( scope.enableRotate === false ) return;
842 
843 				handleTouchStartRotate( event );
844 
845 				state = STATE.TOUCH_ROTATE;
846 
847 				break;
848 
849 			case 2:	// two-fingered touch: dolly-pan
850 
851 				if ( scope.enableZoom === false && scope.enablePan === false ) return;
852 
853 				handleTouchStartDollyPan( event );
854 
855 				state = STATE.TOUCH_DOLLY_PAN;
856 
857 				break;
858 
859 			default:
860 
861 				state = STATE.NONE;
862 
863 		}
864 
865 		if ( state !== STATE.NONE ) {
866 
867 			scope.dispatchEvent( startEvent );
868 
869 		}
870 
871 	}

873 	function onTouchMove( event ) {
874 
875 		if ( scope.enabled === false ) return;
876 
877 		event.preventDefault();
878 		event.stopPropagation();
879 
880 		switch ( event.touches.length ) {
881 
882 			case 1: // one-fingered touch: rotate
883 
884 				if ( scope.enableRotate === false ) return;
885 				if ( state !== STATE.TOUCH_ROTATE ) return; // is this needed?
886 
887 				handleTouchMoveRotate( event );
888 
889 				break;
890 
891 			case 2: // two-fingered touch: dolly-pan
892 
893 				if ( scope.enableZoom === false && scope.enablePan === false ) return;
894 				if ( state !== STATE.TOUCH_DOLLY_PAN ) return; // is this needed?
895 
896 				handleTouchMoveDollyPan( event );
897 
898 				break;
899 
900 			default:
901 
902 				state = STATE.NONE;
903 
904 		}
905 
906 	}
