  6 function earcut(data, holeIndices, dim) {
  7 
  8     dim = dim || 2;
  9 
 10     var hasHoles = holeIndices && holeIndices.length,
 11         outerLen = hasHoles ? holeIndices[0] * dim : data.length,
 12         outerNode = linkedList(data, 0, outerLen, dim, true),
 13         triangles = [];
 14 
 15     if (!outerNode || outerNode.next === outerNode.prev) return triangles;
 16 
 17     var minX, minY, maxX, maxY, x, y, invSize;
 18 
 19     if (hasHoles) outerNode = eliminateHoles(data, holeIndices, outerNode, dim);
 20 
 21     // if the shape is not too simple, we'll use z-order curve hash later; calculate polygon bbox
 22     if (data.length > 80 * dim) {
 23         minX = maxX = data[0];
 24         minY = maxY = data[1];
 25 
 26         for (var i = dim; i < outerLen; i += dim) {
 27             x = data[i];
 28             y = data[i + 1];
 29             if (x < minX) minX = x;
 30             if (y < minY) minY = y;
 31             if (x > maxX) maxX = x;
 32             if (y > maxY) maxY = y;
 33         }
 34 
 35         // minX, minY and invSize are later used to transform coords into integers for z-order calculation
 36         invSize = Math.max(maxX - minX, maxY - minY);
 37         invSize = invSize !== 0 ? 1 / invSize : 0;
 38     }
 39 
 40     earcutLinked(outerNode, triangles, dim, minX, minY, invSize);
 41 
 42     return triangles;
 43 }

 45 // create a circular doubly linked list from polygon points in the specified winding order
 46 function linkedList(data, start, end, dim, clockwise) {
 47     var i, last;
 48 
 49     if (clockwise === (signedArea(data, start, end, dim) > 0)) {
 50         for (i = start; i < end; i += dim) last = insertNode(i, data[i], data[i + 1], last);
 51     } else {
 52         for (i = end - dim; i >= start; i -= dim) last = insertNode(i, data[i], data[i + 1], last);
 53     }
 54 
 55     if (last && equals(last, last.next)) {
 56         removeNode(last);
 57         last = last.next;
 58     }
 59 
 60     return last;
 61 }

 63 // eliminate colinear or duplicate points
 64 function filterPoints(start, end) {
 65     if (!start) return start;
 66     if (!end) end = start;
 67 
 68     var p = start,
 69         again;
 70     do {
 71         again = false;
 72 
 73         if (!p.steiner && (equals(p, p.next) || area(p.prev, p, p.next) === 0)) {
 74             removeNode(p);
 75             p = end = p.prev;
 76             if (p === p.next) break;
 77             again = true;
 78 
 79         } else {
 80             p = p.next;
 81         }
 82     } while (again || p !== end);
 83 
 84     return end;
 85 }

 87 // main ear slicing loop which triangulates a polygon (given as a linked list)
 88 function earcutLinked(ear, triangles, dim, minX, minY, invSize, pass) {
 89     if (!ear) return;
 90 
 91     // interlink polygon nodes in z-order
 92     if (!pass && invSize) indexCurve(ear, minX, minY, invSize);
 93 
 94     var stop = ear,
 95         prev, next;
 96 
 97     // iterate through ears, slicing them one by one
 98     while (ear.prev !== ear.next) {
 99         prev = ear.prev;
100         next = ear.next;
101 
102         if (invSize ? isEarHashed(ear, minX, minY, invSize) : isEar(ear)) {
103             // cut off the triangle
104             triangles.push(prev.i / dim);
105             triangles.push(ear.i / dim);
106             triangles.push(next.i / dim);
107 
108             removeNode(ear);
109 
110             // skipping the next vertex leads to less sliver triangles
111             ear = next.next;
112             stop = next.next;
113 
114             continue;
115         }
116 
117         ear = next;
118 
119         // if we looped through the whole remaining polygon and can't find any more ears
120         if (ear === stop) {
121             // try filtering points and slicing again
122             if (!pass) {
123                 earcutLinked(filterPoints(ear), triangles, dim, minX, minY, invSize, 1);
124 
125             // if this didn't work, try curing all small self-intersections locally
126             } else if (pass === 1) {
127                 ear = cureLocalIntersections(filterPoints(ear), triangles, dim);
128                 earcutLinked(ear, triangles, dim, minX, minY, invSize, 2);
129 
130             // as a last resort, try splitting the remaining polygon into two
131             } else if (pass === 2) {
132                 splitEarcut(ear, triangles, dim, minX, minY, invSize);
133             }
134 
135             break;
136         }
137     }
138 }

140 // check whether a polygon node forms a valid ear with adjacent nodes
141 function isEar(ear) {
142     var a = ear.prev,
143         b = ear,
144         c = ear.next;
145 
146     if (area(a, b, c) >= 0) return false; // reflex, can't be an ear
147 
148     // now make sure we don't have other points inside the potential ear
149     var p = ear.next.next;
150 
151     while (p !== ear.prev) {
152         if (pointInTriangle(a.x, a.y, b.x, b.y, c.x, c.y, p.x, p.y) &&
153             area(p.prev, p, p.next) >= 0) return false;
154         p = p.next;
155     }
156 
157     return true;
158 }

160 function isEarHashed(ear, minX, minY, invSize) {
161     var a = ear.prev,
162         b = ear,
163         c = ear.next;
164 
165     if (area(a, b, c) >= 0) return false; // reflex, can't be an ear
166 
167     // triangle bbox; min & max are calculated like this for speed
168     var minTX = a.x < b.x ? (a.x < c.x ? a.x : c.x) : (b.x < c.x ? b.x : c.x),
169         minTY = a.y < b.y ? (a.y < c.y ? a.y : c.y) : (b.y < c.y ? b.y : c.y),
170         maxTX = a.x > b.x ? (a.x > c.x ? a.x : c.x) : (b.x > c.x ? b.x : c.x),
171         maxTY = a.y > b.y ? (a.y > c.y ? a.y : c.y) : (b.y > c.y ? b.y : c.y);
172 
173     // z-order range for the current triangle bbox;
174     var minZ = zOrder(minTX, minTY, minX, minY, invSize),
175         maxZ = zOrder(maxTX, maxTY, minX, minY, invSize);
176 
177     var p = ear.prevZ,
178         n = ear.nextZ;
179 
180     // look for points inside the triangle in both directions
181     while (p && p.z >= minZ && n && n.z <= maxZ) {
182         if (p !== ear.prev && p !== ear.next &&
183             pointInTriangle(a.x, a.y, b.x, b.y, c.x, c.y, p.x, p.y) &&
184             area(p.prev, p, p.next) >= 0) return false;
185         p = p.prevZ;
186 
187         if (n !== ear.prev && n !== ear.next &&
188             pointInTriangle(a.x, a.y, b.x, b.y, c.x, c.y, n.x, n.y) &&
189             area(n.prev, n, n.next) >= 0) return false;
190         n = n.nextZ;
191     }
192 
193     // look for remaining points in decreasing z-order
194     while (p && p.z >= minZ) {
195         if (p !== ear.prev && p !== ear.next &&
196             pointInTriangle(a.x, a.y, b.x, b.y, c.x, c.y, p.x, p.y) &&
197             area(p.prev, p, p.next) >= 0) return false;
198         p = p.prevZ;
199     }
200 
201     // look for remaining points in increasing z-order
202     while (n && n.z <= maxZ) {
203         if (n !== ear.prev && n !== ear.next &&
204             pointInTriangle(a.x, a.y, b.x, b.y, c.x, c.y, n.x, n.y) &&
205             area(n.prev, n, n.next) >= 0) return false;
206         n = n.nextZ;
207     }
208 
209     return true;
210 }

212 // go through all polygon nodes and cure small local self-intersections
213 function cureLocalIntersections(start, triangles, dim) {
214     var p = start;
215     do {
216         var a = p.prev,
217             b = p.next.next;
218 
219         if (!equals(a, b) && intersects(a, p, p.next, b) && locallyInside(a, b) && locallyInside(b, a)) {
220 
221             triangles.push(a.i / dim);
222             triangles.push(p.i / dim);
223             triangles.push(b.i / dim);
224 
225             // remove two nodes involved
226             removeNode(p);
227             removeNode(p.next);
228 
229             p = start = b;
230         }
231         p = p.next;
232     } while (p !== start);
233 
234     return filterPoints(p);
235 }

237 // try splitting polygon into two and triangulate them independently
238 function splitEarcut(start, triangles, dim, minX, minY, invSize) {
239     // look for a valid diagonal that divides the polygon into two
240     var a = start;
241     do {
242         var b = a.next.next;
243         while (b !== a.prev) {
244             if (a.i !== b.i && isValidDiagonal(a, b)) {
245                 // split the polygon in two by the diagonal
246                 var c = splitPolygon(a, b);
247 
248                 // filter colinear points around the cuts
249                 a = filterPoints(a, a.next);
250                 c = filterPoints(c, c.next);
251 
252                 // run earcut on each half
253                 earcutLinked(a, triangles, dim, minX, minY, invSize);
254                 earcutLinked(c, triangles, dim, minX, minY, invSize);
255                 return;
256             }
257             b = b.next;
258         }
259         a = a.next;
260     } while (a !== start);
261 }

263 // link every hole into the outer loop, producing a single-ring polygon without holes
264 function eliminateHoles(data, holeIndices, outerNode, dim) {
265     var queue = [],
266         i, len, start, end, list;
267 
268     for (i = 0, len = holeIndices.length; i < len; i++) {
269         start = holeIndices[i] * dim;
270         end = i < len - 1 ? holeIndices[i + 1] * dim : data.length;
271         list = linkedList(data, start, end, dim, false);
272         if (list === list.next) list.steiner = true;
273         queue.push(getLeftmost(list));
274     }
275 
276     queue.sort(compareX);
277 
278     // process holes from left to right
279     for (i = 0; i < queue.length; i++) {
280         eliminateHole(queue[i], outerNode);
281         outerNode = filterPoints(outerNode, outerNode.next);
282     }
283 
284     return outerNode;
285 }

291 // find a bridge between vertices that connects hole with an outer ring and and link it
292 function eliminateHole(hole, outerNode) {
293     outerNode = findHoleBridge(hole, outerNode);
294     if (outerNode) {
295         var b = splitPolygon(outerNode, hole);
296 
297         // filter collinear points around the cuts
298         filterPoints(outerNode, outerNode.next);
299         filterPoints(b, b.next);
300     }
301 }

303 // David Eberly's algorithm for finding a bridge between hole and outer polygon
304 function findHoleBridge(hole, outerNode) {
305     var p = outerNode,
306         hx = hole.x,
307         hy = hole.y,
308         qx = -Infinity,
309         m;
310 
311     // find a segment intersected by a ray from the hole's leftmost point to the left;
312     // segment's endpoint with lesser x will be potential connection point
313     do {
314         if (hy <= p.y && hy >= p.next.y && p.next.y !== p.y) {
315             var x = p.x + (hy - p.y) * (p.next.x - p.x) / (p.next.y - p.y);
316             if (x <= hx && x > qx) {
317                 qx = x;
318                 if (x === hx) {
319                     if (hy === p.y) return p;
320                     if (hy === p.next.y) return p.next;
321                 }
322                 m = p.x < p.next.x ? p : p.next;
323             }
324         }
325         p = p.next;
326     } while (p !== outerNode);
327 
328     if (!m) return null;
329 
330     if (hx === qx) return m; // hole touches outer segment; pick leftmost endpoint
331 
332     // look for points inside the triangle of hole point, segment intersection and endpoint;
333     // if there are no points found, we have a valid connection;
334     // otherwise choose the point of the minimum angle with the ray as connection point
335 
336     var stop = m,
337         mx = m.x,
338         my = m.y,
339         tanMin = Infinity,
340         tan;
341 
342     p = m;
343 
344     do {
345         if (hx >= p.x && p.x >= mx && hx !== p.x &&
346                 pointInTriangle(hy < my ? hx : qx, hy, mx, my, hy < my ? qx : hx, hy, p.x, p.y)) {
347 
348             tan = Math.abs(hy - p.y) / (hx - p.x); // tangential
349 
350             if (locallyInside(p, hole) &&
351                 (tan < tanMin || (tan === tanMin && (p.x > m.x || (p.x === m.x && sectorContainsSector(m, p)))))) {
352                 m = p;
353                 tanMin = tan;
354             }
355         }
356 
357         p = p.next;
358     } while (p !== stop);
359 
360     return m;
361 }

363 // whether sector in vertex m contains sector in vertex p in the same coordinates
364 function sectorContainsSector(m, p) {
365     return area(m.prev, m, p.prev) < 0 && area(p.next, m, m.next) < 0;
366 }

368 // interlink polygon nodes in z-order
369 function indexCurve(start, minX, minY, invSize) {
370     var p = start;
371     do {
372         if (p.z === null) p.z = zOrder(p.x, p.y, minX, minY, invSize);
373         p.prevZ = p.prev;
374         p.nextZ = p.next;
375         p = p.next;
376     } while (p !== start);
377 
378     p.prevZ.nextZ = null;
379     p.prevZ = null;
380 
381     sortLinked(p);
382 }

384 // Simon Tatham's linked list merge sort algorithm
385 // http://www.chiark.greenend.org.uk/~sgtatham/algorithms/listsort.html
386 function sortLinked(list) {
387     var i, p, q, e, tail, numMerges, pSize, qSize,
388         inSize = 1;
389 
390     do {
391         p = list;
392         list = null;
393         tail = null;
394         numMerges = 0;
395 
396         while (p) {
397             numMerges++;
398             q = p;
399             pSize = 0;
400             for (i = 0; i < inSize; i++) {
401                 pSize++;
402                 q = q.nextZ;
403                 if (!q) break;
404             }
405             qSize = inSize;
406 
407             while (pSize > 0 || (qSize > 0 && q)) {
408 
409                 if (pSize !== 0 && (qSize === 0 || !q || p.z <= q.z)) {
410                     e = p;
411                     p = p.nextZ;
412                     pSize--;
413                 } else {
414                     e = q;
415                     q = q.nextZ;
416                     qSize--;
417                 }
418 
419                 if (tail) tail.nextZ = e;
420                 else list = e;
421 
422                 e.prevZ = tail;
423                 tail = e;
424             }
425 
426             p = q;
427         }
428 
429         tail.nextZ = null;
430         inSize *= 2;
431 
432     } while (numMerges > 1);
433 
434     return list;
435 }

437 // z-order of a point given coords and inverse of the longer side of data bbox
438 function zOrder(x, y, minX, minY, invSize) {
439     // coords are transformed into non-negative 15-bit integer range
440     x = 32767 * (x - minX) * invSize;
441     y = 32767 * (y - minY) * invSize;
442 
443     x = (x | (x << 8)) & 0x00FF00FF;
444     x = (x | (x << 4)) & 0x0F0F0F0F;
445     x = (x | (x << 2)) & 0x33333333;
446     x = (x | (x << 1)) & 0x55555555;
447 
448     y = (y | (y << 8)) & 0x00FF00FF;
449     y = (y | (y << 4)) & 0x0F0F0F0F;
450     y = (y | (y << 2)) & 0x33333333;
451     y = (y | (y << 1)) & 0x55555555;
452 
453     return x | (y << 1);
454 }

456 // find the leftmost node of a polygon ring
457 function getLeftmost(start) {
458     var p = start,
459         leftmost = start;
460     do {
461         if (p.x < leftmost.x || (p.x === leftmost.x && p.y < leftmost.y)) leftmost = p;
462         p = p.next;
463     } while (p !== start);
464 
465     return leftmost;

468 // check if a point lies within a convex triangle
469 function pointInTriangle(ax, ay, bx, by, cx, cy, px, py) {
470     return (cx - px) * (ay - py) - (ax - px) * (cy - py) >= 0 &&
471            (ax - px) * (by - py) - (bx - px) * (ay - py) >= 0 &&
472            (bx - px) * (cy - py) - (cx - px) * (by - py) >= 0;
473 }

475 // check if a diagonal between two polygon nodes is valid (lies in polygon interior)
476 function isValidDiagonal(a, b) {
477     return a.next.i !== b.i && a.prev.i !== b.i && !intersectsPolygon(a, b) && // dones't intersect other edges
478            (locallyInside(a, b) && locallyInside(b, a) && middleInside(a, b) && // locally visible
479             (area(a.prev, a, b.prev) || area(a, b.prev, b)) || // does not create opposite-facing sectors
480             equals(a, b) && area(a.prev, a, a.next) > 0 && area(b.prev, b, b.next) > 0); // special zero-length case
481 }

483 // signed area of a triangle
484 function area(p, q, r) {
485     return (q.y - p.y) * (r.x - q.x) - (q.x - p.x) * (r.y - q.y);
486 }

489 function equals(p1, p2) {
490     return p1.x === p2.x && p1.y === p2.y;

493 // check if two segments intersect
494 function intersects(p1, q1, p2, q2) {
495     var o1 = sign(area(p1, q1, p2));
496     var o2 = sign(area(p1, q1, q2));
497     var o3 = sign(area(p2, q2, p1));
498     var o4 = sign(area(p2, q2, q1));
499 
500     if (o1 !== o2 && o3 !== o4) return true; // general case
501 
502     if (o1 === 0 && onSegment(p1, p2, q1)) return true; // p1, q1 and p2 are collinear and p2 lies on p1q1
503     if (o2 === 0 && onSegment(p1, q2, q1)) return true; // p1, q1 and q2 are collinear and q2 lies on p1q1
504     if (o3 === 0 && onSegment(p2, p1, q2)) return true; // p2, q2 and p1 are collinear and p1 lies on p2q2
505     if (o4 === 0 && onSegment(p2, q1, q2)) return true; // p2, q2 and q1 are collinear and q1 lies on p2q2
506 
507     return false;
508 }

510 // for collinear points p, q, r, check if point q lies on segment pr
511 function onSegment(p, q, r) {
512     return q.x <= Math.max(p.x, r.x) && q.x >= Math.min(p.x, r.x) && q.y <= Math.max(p.y, r.y) && q.y >= Math.min(p.y, r.y);
513 }

516     return num > 0 ? 1 : num < 0 ? -1 : 0;

519 // check if a polygon diagonal intersects any polygon segments
520 function intersectsPolygon(a, b) {
521     var p = a;
522     do {
523         if (p.i !== a.i && p.next.i !== a.i && p.i !== b.i && p.next.i !== b.i &&
524                 intersects(p, p.next, a, b)) return true;
525         p = p.next;
526     } while (p !== a);
527 
528     return false;
529 }

531 // check if a polygon diagonal is locally inside the polygon
532 function locallyInside(a, b) {
533     return area(a.prev, a, a.next) < 0 ?
534         area(a, b, a.next) >= 0 && area(a, a.prev, b) >= 0 :
535         area(a, b, a.prev) < 0 || area(a, a.next, b) < 0;
536 }

538 // check if the middle point of a polygon diagonal is inside the polygon
539 function middleInside(a, b) {
540     var p = a,
541         inside = false,
542         px = (a.x + b.x) / 2,
543         py = (a.y + b.y) / 2;
544     do {
545         if (((p.y > py) !== (p.next.y > py)) && p.next.y !== p.y &&
546                 (px < (p.next.x - p.x) * (py - p.y) / (p.next.y - p.y) + p.x))
547             inside = !inside;
548         p = p.next;
549     } while (p !== a);
550 
551     return inside;
552 }

554 // link two polygon vertices with a bridge; if the vertices belong to the same ring, it splits polygon into two;
555 // if one belongs to the outer ring and another to a hole, it merges it into a single ring
556 function splitPolygon(a, b) {
557     var a2 = new Node(a.i, a.x, a.y),
558         b2 = new Node(b.i, b.x, b.y),
559         an = a.next,
560         bp = b.prev;
561 
562     a.next = b;
563     b.prev = a;
564 
565     a2.next = an;
566     an.prev = a2;
567 
568     b2.next = a2;
569     a2.prev = b2;
570 
571     bp.next = b2;
572     b2.prev = bp;
573 
574     return b2;
575 }

577 // create a node and optionally link it with previous one (in a circular doubly linked list)
578 function insertNode(i, x, y, last) {
579     var p = new Node(i, x, y);
580 
581     if (!last) {
582         p.prev = p;
583         p.next = p;
584 
585     } else {
586         p.next = last.next;
587         p.prev = last;
588         last.next.prev = p;
589         last.next = p;
590     }
591     return p;
592 }

594 function removeNode(p) {
595     p.next.prev = p.prev;
596     p.prev.next = p.next;
597 
598     if (p.prevZ) p.prevZ.nextZ = p.nextZ;
599     if (p.nextZ) p.nextZ.prevZ = p.prevZ;
600 }

602 function Node(i, x, y) {
603     // vertex index in coordinates array
604     this.i = i;
605 
606     // vertex coordinates
607     this.x = x;
608     this.y = y;
609 
610     // previous and next vertex nodes in a polygon ring
611     this.prev = null;
612     this.next = null;
613

615     this.z = null;
616

618     this.prevZ = null;
619     this.nextZ = null;
620

622     this.steiner = false;

625 // return a percentage difference between the polygon area and its triangulation area;
626 // used to verify correctness of triangulation
627 earcut.deviation = function (data, holeIndices, dim, triangles) {
628     var hasHoles = holeIndices && holeIndices.length;
629     var outerLen = hasHoles ? holeIndices[0] * dim : data.length;
630 
631     var polygonArea = Math.abs(signedArea(data, 0, outerLen, dim));
632     if (hasHoles) {
633         for (var i = 0, len = holeIndices.length; i < len; i++) {
634             var start = holeIndices[i] * dim;
635             var end = i < len - 1 ? holeIndices[i + 1] * dim : data.length;
636             polygonArea -= Math.abs(signedArea(data, start, end, dim));
637         }
638     }
639 
640     var trianglesArea = 0;
641     for (i = 0; i < triangles.length; i += 3) {
642         var a = triangles[i] * dim;
643         var b = triangles[i + 1] * dim;
644         var c = triangles[i + 2] * dim;
645         trianglesArea += Math.abs(
646             (data[a] - data[c]) * (data[b + 1] - data[a + 1]) -
647             (data[a] - data[b]) * (data[c + 1] - data[a + 1]));
648     }
649 
650     return polygonArea === 0 && trianglesArea === 0 ? 0 :
651         Math.abs((trianglesArea - polygonArea) / polygonArea);
652 };

654 function signedArea(data, start, end, dim) {
655     var sum = 0;
656     for (var i = start, j = end - dim; i < end; i += dim) {
657         sum += (data[j] - data[i]) * (data[i + 1] + data[j + 1]);
658         j = i;
659     }
660     return sum;
661 }

663 // turn a polygon in a multi-dimensional array form (e.g. as in GeoJSON) into a form Earcut accepts
664 earcut.flatten = function (data) {
665     var dim = data[0][0].length,
666         result = {vertices: [], holes: [], dimensions: dim},
667         holeIndex = 0;
668 
669     for (var i = 0; i < data.length; i++) {
670         for (var j = 0; j < data[i].length; j++) {
671             for (var d = 0; d < dim; d++) result.vertices.push(data[i][j][d]);
672         }
673         if (i > 0) {
674             holeIndex += data[i - 1].length;
675             result.holes.push(holeIndex);
676         }
677     }
678     return result;
679 };
