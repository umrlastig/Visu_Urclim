 49 function onMouseMove(evt) {
 50   rect = canvas.getBoundingClientRect();
 51   // calculate mouse position in normalized device coordinates
 52   mouse.x = ((evt.clientX - rect.left) / (rect.right - rect.left)) * 2 - 1;

124 export class xzGrid {
125   constructor(xmin, xmax, xstep, zmin, zmax, zstep) {
126     this.centers = []; // holds the xz coordinates of the grid nodes
127     this.n = Math.floor((xmax - xmin) / xstep);
128     this.m = Math.floor((zmax - zmin) / zstep);
129     for (let i = 0; i < this.n; i++) {
130       for (let j = 0; j < this.m; j++) {
131         this.centers.push({ x: xmin + i * xstep, z: zmin + j * zstep });
132       }
133     }
134   }
135 
136   at(i, j) {
137     return this.centers[i * n + j];
138   }
139 }

141 class Visualisation {
142   constructor() {
143     if (new.target === Visualisation) {
144       throw new TypeError("Cannot construct Visualisation instances directly.");
145     }
146     this.data = []; // collection of {x, y, z, t} objects
147     this.group = new THREE.Group();
148     this.meshes = [];
149     this.loaded = false; // hack
150   }
151   /**
152    *
153    * @param {THREE.Vector3} center
154    */
155   centerOn(center = new THREE.Vector3(0, 0, 0)) {
156     let avg_x = _.meanBy(this.meshes, (mesh) => mesh.position.x);
157     let avg_y = _.meanBy(this.meshes, (mesh) => mesh.position.y);
158     let avg_z = _.meanBy(this.meshes, (mesh) => mesh.position.z);
159     let avg_pos = new THREE.Vector3(avg_x, avg_y, avg_z);
160     this.meshes.forEach((mesh) => {
161       mesh.position.sub(avg_pos).add(center);
162     });
163   }
164 
165   /**
166    *
167    * @param {THREE.Scene} scene
168    */
169   show(scene) {
170     scene.add(...this.groups);
171   }
172 
173   /**
174    *
175    * @param {THREE.Scene} scene
176    */
177   hide(scene) {
178     scene.remove(...this.groups);
179   }
180 }

182 export class BarVisualisation extends Visualisation {
183   /**
184    *
185    * @param {*} data
186    * @param {*} xAttribute
187    * @param {*} yAttribute
188    * @param {*} zAttribute
189    *
190    * Builds a bar visualisation of the data array
191    */
192   constructor(data, xAttribute, yAttribute, zAttribute, grid, config) {
193     super();
194     this.data = data;
195     this.colors = {};
196     this.edges = [];
197     this.xAttribute = xAttribute;
198     this.yAttribute = yAttribute;
199     this.zAttribute = zAttribute;
200     this.grid = grid;
201     this.groups = [];
202     this.axisData = [];
203     this.data.forEach((datael, index) => {
204       let obj = {
205         x: datael[xAttribute],
206         y: datael[yAttribute],
207         z: datael[zAttribute],
208       };
209       if (config.extraAttributes) {
210         for (let attribute of config.extraAttributes) {
211           obj[attribute] = datael[attribute];
212         }
213       }
214       this.axisData.push(obj);
215     });
216     this.ymin = getMin(this.axisData.map((el) => el.y));
217     this.ymax = getMax(this.axisData.map((el) => el.y));
218     let yscale;
219     if (config.yscale) {
220       yscale = config.yscale;
221     } else {
222       yscale = 1 / (this.ymax - this.ymin);
223     }
224     let yoffset;
225     if (config.yoffset) {
226       yoffset = -this.ymin * yscale + 0.0001;
227     } else {
228       yoffset = 0;
229     }
230     
231     this.axisData.forEach((e) => {
232       let geom = new THREE.BoxGeometry(
233         this.step,
234         e.y * yscale + yoffset,
235         this.step
236       );
237       let material;
238       this.highLightOffset = new THREE.Color(1, 1, 1).sub(new THREE.Color(attributeColor(1))).multiplyScalar(0.5);
239       let color;
240       if (config.material) {
241         material = config.material;
242       } else {
243         color = attributeColor((e.y - this.ymin) / (this.ymax - this.ymin));
244         material = new THREE.MeshStandardMaterial({
245           color,
246           transparent: true,
247           opacity: 1,
248         });
249       }
250 
251       let shape = new THREE.Mesh(geom, material);
252       shape.data = e;
253       shape.originalColor = color;
254       shape.resetColor = function () {
255         this.material.color = new THREE.Color(this.originalColor);
256       };
257       let edges = new THREE.EdgesGeometry(geom);
258       const line = new THREE.LineSegments(
259         edges,
260         new THREE.LineBasicMaterial({ 
261             color: 0xffffff,
262             linewidth: 3
263         })
264       );
265 
266       shape.position.set(e.x, (e.y * yscale + yoffset) / 2, e.z);
267       shape.line = line;
268       shape.showLine = function () {
269         shape.add(this.line);
270       };
271       shape.removeLine = function () {
272         shape.remove(...shape.children);
273       };
274       this.meshes.push(shape);
275       this.loaded = true;
276     });
277   }
278   /**
279    * Group meshes by axis
280    * @param {String} axis set the axis
281    */
282   setgroupAxis(axis) {
283     let groupLists = _.groupBy(this.meshes, (mesh) => {
284       return mesh.position[axis];
285     });
286     for (let groupList in groupLists) {
287       let group = new THREE.Group();
288       for (let mesh of groupLists[groupList]) {
289         group.add(mesh);
290       }
291       this.groups.push(group);
292     }
293     this.group = new THREE.Group();
294     this.group.add(...this.groups);
295   }
296 
297   highlightGroup(group) {
298     this.resetColors();
299     for (let mesh of group.children) {
300       if (mesh.material.color) mesh.material.color.add(this.highLightOffset);
301     }
302   }
303 
304   selectGroup(group) {
305     this.unselectAllGroups();
306     for (let mesh of group.children) {
307       mesh.showLine();
308     }
309   }
310 
311   unselectAllGroups() {
312     for (let group of this.groups) {
313       for (let mesh of group.children) {
314         mesh.removeLine();
315       }
316     }
317   }
318 
319   resetColors() {
320     for (let group of this.groups) {
321       for (let mesh of group.children) {
322         mesh.resetColor();
323       }
324     }
325   }
326 }

340 /**
341  * This function attribute the good color to each block depending on it temperature
342  * @param {*} temp
343  * @returns
344  */
345  function attributeColor(temp) {
346   var color_array = [
347     "#39519E",
348     "#3C60A4",
349     "#5C8DC0",
350     "#8FC1DC",
351     "#ACD7E7",
352     "#DFE5CD",
353     "#FCC97D",
354     "#FCAC64",
355     "#F5844D",
356     "#CD322C",
357   ];
358 
359   let id = temp*10
360 
361   let under = Math.floor(temp / 0.1);
362   let over = Math.ceil(temp / 0.1);
363 
364   if (over == 10) {
365       over = 9
366   }
367   if (id == 10) {
368       return color_array[9]
369   }
370 
371 
372 
373   let color = (new THREE.Color(color_array[under])).lerp(new THREE.Color(color_array[over]), id-under)
374   return color.getHex()
375 }

415 let dateViewer = document.getElementById("dateViewer");
416 async function onHoverPick() {
417 
418       // update the picking ray with the camera and mouse position
419   raycaster.setFromCamera(mouse, camera);
420 
421   // calculate objects intersecting the picking ray
422   const intersects = raycaster
423     .intersectObjects(scene.children, true)
424     .filter((intersect) => intersect.object.type === "Mesh");
425 
426   if (intersects.length) {
427     let object = intersects[0].object;
428     visu.highlightGroup(object.parent);
429     dateViewer.innerText = object.data.date;
430   } else {
431     dateViewer.innerText = "";
432     visu.resetColors();
433   }
434 }
