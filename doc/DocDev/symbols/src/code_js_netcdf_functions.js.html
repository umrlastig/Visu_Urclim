 52 /**
 53  * Cette fonction permet d'ouvrir et de lire un fichier netcdf
 54  * @param {*} response 
 55  * @returns un buffer netcdf
 56  */
 57 export function readNetcdf(response) {
 58   var reader = response.body.getReader();
 59   var bytesReceived = 0;
 60   var buffer = null;
 61   return reader.read().then(function process(result) {
 62     // append new bytes to the buffer
 63     if (!result.done) {
 64       var received = bytesReceived + result.value.length;
 65     }
 66     if (!buffer) {
 67       buffer = result.value;
 68     } else {
 69       var old = buffer.subarray(0, bytesReceived);
 70       buffer = new Uint8Array(received);
 71       buffer.set(old, 0, bytesReceived);
 72       buffer.set(result.value, bytesReceived);
 73     }
 74     bytesReceived = received;
 75 
 76     // try reading the header
 77     try {
 78       const netcdf = new netcdfjs(buffer);
 79       netcdf.url = response.url;
 80       netcdf.reader = reader;
 81       netcdf.bytesTotal = response.headers.get("Content-Length");
 82       netcdf.acceptRanges = response.headers.get("Accept-Ranges") === "bytes";
 83       netcdf.bytesReceived = bytesReceived;
 84       return netcdf;
 85     } catch (e) {
 86       // end is reached with no decodable header
 87       if (result.done) {
 88         console.log("eof, no valid netcdf header!");
 89         return;
 90       }
 91       // keep reading
 92       return reader.read().then(process);
 93     }
 94   });
 95 }

 97 /**
 98  * Cette fonction permet de recuperer les données propre au type étudié en les stockant dans un objet volume
 99  * @param {*} netcdf 
100  * @param {*} variableName 
101  * @param {*} forceRangeRequest 
102  * @returns lance la onction decodeVolume
103  */
104 export async function fetchVolume(
105   netcdf,
106   variableName,
107   forceRangeRequest = false
108 ) {
109   const rangeRequest = forceRangeRequest || netcdf.acceptRanges;
110   var variable = netcdf.variables.find((val) => val.name === variableName);
111 
112   if (variableName == "THT") {
113     var volume = {
114       variable: variableName,
115       xLength: netcdf.dimensions[variable.dimensions[1]].size,
116       yLength: netcdf.dimensions[variable.dimensions[2]].size,
117       zLength: netcdf.dimensions[variable.dimensions[3]].size,
118     };
119   } else if (variableName == "ZHAT" ) {
120     var volume = {
121       variable: variableName,
122       xLength: 1,
123       yLength: netcdf.dimensions[variable.dimensions[0]].size,
124       zLength: 1,
125     };
126   }else if(variableName == "DTCUR"){
127     
128     var volume = {
129       variable: variableName,
130       variableType: variable.attributes[2].value.substring(14,24), 
131       xLength: 1,
132       yLength: 1,
133       zLength: 1,
134     };
135   } else {
136     var volume = {
137       variable: variableName,
138       xLength: 1,
139       yLength: netcdf.dimensions[variable.dimensions[0]].size,
140       zLength: netcdf.dimensions[variable.dimensions[1]].size,
141     };
142   }
143 
144   volume.size = volume.xLength * volume.yLength * volume.zLength;
145   const last = variable.offset + variable.size - 1;
146 
147   // // Data is missing and ranges are not accepted
148   if (!rangeRequest && last >= netcdf.bytesReceived) {
149     // TODO, use reader if present
150     if (netcdf.reader) {
151       // TODO, use reader if present
152     }
153 
154     // TODO, stop if enough bytes are read
155     return fetch(netcdf.url)
156       .then((response) => response.arrayBuffer())
157       .then((buffer) => decodeVolume(buffer, volume, variable.offset));
158   }
159 
160   if (netcdf.reader) {
161     netcdf.reader.cancel();
162     netcdf.reader = undefined;
163   }
164 
165   if (last < netcdf.bytesReceived)
166     return decodeVolume(netcdf.buffer.buffer, volume, variable.offset);
167 
168   // Data is missing, get it using a range request
169   const headers = new Headers({
170     Range: `bytes=${variable.offset}-${last}`,
171   });
172   return fetch(netcdf.url, { headers })
173     .then((response) => response.arrayBuffer())
174     .then((buffer) => decodeVolume(buffer, volume));
175 }

178 /**
179  * Cette fonction permet de donner du sens aux valeurs conenues dans un volume
180  * @param {*} buffer 
181  * @param {*} volume 
182  * @param {*} offset 
183  * @returns un array contenant la valeur de chaque point pour le type étudié.
184  */
185 async function decodeVolume(buffer, volume, offset = 0) {
186   volume.data = new Float32Array(volume.size);
187   var view = new DataView(buffer);
188   for (var i = 0; i < volume.size; i++)
189     volume.data[i] = view.getFloat64(offset + i * 8, false);
190   volume.min = Math.min.apply(
191     null,
192     volume.data.filter((x) => x > 1)
193   );
194   volume.max = Math.max.apply(null, volume.data);
195   return volume;
196 }
