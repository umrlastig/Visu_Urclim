  4 var tailleGraph;

 11 curseur.addEventListener('mousedown', e => {

 16 window.addEventListener('mousemove', e => {
 17     if (bool === true) {
 18         change(posY)
 19         posY = e.clientY;
 20     }
 21 });
 22 
 23 window.addEventListener('mouseup', e => {
 24     if (bool === true) {
 25         change(posY);
 26         posY = 0;
 27         bool = false;
 28     }

 31 /**
 32  * Cette fonction permet de modifier les tailles des deux fenetres de visualisation
 33  * @param {*} posY 
 34  */
 35 export function change(posY) {
 36     cont.style.height = posY + "px";
 37     tailleGraph = heightScreen - 10 - posY;
 38     graph.style.height = tailleGraph + "px";
 39   }

 50 /**
 51  * Cette fonction permet de switcher de tableau entre le mode 2D et 3D
 52  * @param {*} evt 
 53  * @param {*} graph 
 54  */
 55 function showGraph(evt, graph) {
 56     var i, tabcontent, tablinks;
 57     tabcontent = document.getElementsByClassName("tabcontent");
 58     for (i = 0; i < tabcontent.length; i++) {
 59         tabcontent[i].style.display = "none";
 60     }
 61     tablinks = document.getElementsByClassName("tablinks");
 62     for (i = 0; i < tablinks.length; i++) {
 63         tablinks[i].className = tablinks[i].className.replace(" active", "");
 64     }
 65     document.getElementById(graph).style.display = "block";
 66     evt.currentTarget.className += " active";
 67 }
