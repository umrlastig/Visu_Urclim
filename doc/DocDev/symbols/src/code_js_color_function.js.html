 29 function hexToRgb(hex) {
 30 	  var result = /^#?([a-f\d]{2})([a-f\d]{2})([a-f\d]{2})$/i.exec(hex);
 31 	  return result ? {
 32 		r: parseInt(result[1], 16),
 33 		g: parseInt(result[2], 16),
 34 		b: parseInt(result[3], 16)
 35 	  } : null;
 36 	}

 38 	function approximateColor1ToColor2ByPercent(color1, color2, percent) {
 39 	
 40 	  var red1 = parseInt(color1[1] + color1[2], 16);
 41 	  var green1 = parseInt(color1[3] + color1[4], 16);
 42 	  var blue1 = parseInt(color1[5] + color1[6], 16);
 43 
 44 	  var red2 = parseInt(color2[1] + color2[2], 16);
 45 	  var green2 = parseInt(color2[3] + color2[4], 16);
 46 	  var blue2 = parseInt(color2[5] + color2[6], 16);
 47 
 48 	  var red = Math.round(mix(red1, red2, percent));
 49 	  var green = Math.round(mix(green1, green2, percent));
 50 	  var blue = Math.round(mix(blue1, blue2, percent));
 51 
 52 	  return generateHex(red, green, blue);
 53 	}

 55 	function generateHex(r, g, b) {
 56 	  r = r.toString(16);
 57 	  g = g.toString(16);
 58 	  b = b.toString(16);
 59 
 60 	  while (r.length < 2) { r = "0" + r; }
 61 	  while (g.length < 2) { g = "0" + g; }
 62 	  while (b.length < 2) { b = "0" + b; }
 63 
 64 	  return "#" + r + g + b;
 65 	}
 66 
 67 	function mix(start, end, percent) {
 68 		return start + ((percent) * (end - start));
 69 	}
