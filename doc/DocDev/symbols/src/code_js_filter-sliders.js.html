  5 // filter = TEMPERATURE SLIDER
  6 $( function() {
  7   let tempMin = general_config.temp_array[0]-273.15;
  8   let tempMax = general_config.temp_array[1]-273.15;
  9   $( "#temp-slider-range" ).slider({
 10     range: true,
 11     min: 0,
 12     max: 100,
 13     values: [ 0, 100 ],
 14     slide: function( event, ui ) {
 15       tempMin = general_config.temp_array[0]-273.15
 16       tempMax = general_config.temp_array[1]-273.15
 17       let tempMinCalc = tempMin + ((tempMax - tempMin)*(ui.values[ 0 ]/100))
 18       let tempMaxCalc = tempMin + ((tempMax - tempMin)*(ui.values[ 1 ]/100))
 19       $( "#temperatures-label" ).val( tempMinCalc.toFixed(2) + "°C - " + tempMaxCalc.toFixed(2) + "°C" );    
 20       general_config.temp_min_factor = (ui.values[ 0 ]/100);  
 21       general_config.temp_max_factor = (ui.values[ 1 ]/100);
 22       
 23     },
 24     stop: function (e, ui) {
 25       recreate_scene()
 26     }
 27   });
 28 
 29   $( "#temperatures-label" ).val(`${tempMin.toFixed(2)}°C - ${tempMax.toFixed(2)}°C`)
 30     
 31 });
